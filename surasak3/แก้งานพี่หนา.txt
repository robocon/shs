    select * from opcardchk where part = 'แขวงทางหลวง1'

    select * from resulthead where orderdate like '2019-09-30%' and clinicalinfo like 'ตรวจสุขภาพประจำปี%' 
) as b on b.hn = a.HN 
where b.autonumber is not null
